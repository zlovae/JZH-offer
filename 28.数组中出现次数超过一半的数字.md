public class Solution {
    public int MoreThanHalfNum_Solution(int [] array) {
        if(array.length==1) return array[0];
        int ans=array[0];
        int times=1;
        
        for(int i : array) {
            if(ans==i) {
                times++;
            } else if(times==0) {
                ans=i;
                times=1;
            } else {
                times--;
            }
        }
        times=0;
        for(int i : array) {
            if(i==ans) {
                times++;
            }
        }
        if(times>(array.length/2))
            return ans;
        return 0;
    }
}
